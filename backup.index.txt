                    <option value="livre">Criar Documento / Livre Personalizado</option>
                </select>

    <script>
        //Funcao para esconder o campo caso seja informado outro tipo de documento
        function checkFormulario() {
            const select = document.getElementById('opcao_formulario');
            const grupoTextoLivre = document.getElementById('grupoTextoLivre');

            if (select.value === 'livre') {
                grupoTextoLivre.classList.remove('hidden');
            } else {
                grupoTextoLivre.classList.add('hidden');
            }
        };

        //Conexão com envio do formulário e conecta com Back-end Flask
        document.getElementById('formGerador').addEventListener('submit', async (e) => {
            // e.preventDefault();

            const divResultado = document.getElementById('resultado');
            divResultado.className = 'hidden';

            const dadosPayload = {
                opcao_formulario: document.getElementById('opcao_formulario').value,
                texto_livre: document.getElementById('texto_livre').value,
                pessoa: {
                    nome: document.getElementById('nome').value,
                    cpf: document.getElementById('cpf').value,
                    endereco: document.getElementById('endereco').value,
                    filho: document.getElementById('filho').value,
                    renda: document.getElementById('renda').value,
                }
            };


            try {
                const resposta = await fetch('http://127.0.0.1:5000/gerar-documento', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(dadosPayload)
                });

                const dadosRetorno = await resposta.json();

                if (resposta.ok) {
                    document.getElementById('textoDoDocumentoGerado').innerText = dadosRetorno.documento;
                    document.getElementById('areaFormulario').style.display = 'none';
                    const areaDoc = document.getElementById('areaDocumento');
                    areaDoc.classList.remove('hidden');


                    // document.body.innerHTML = `
                    // <div class="btn-acoes">
                    //     <button onclick="window.print()"> Salvar em PDF / Imprimir</button>
                    //     <button class="btn-secundario" onclick="window.location.reload()"> Gerar novo documento</button>
                    //     </div>
                    //     <div class="documento-wrapper">
                    //         <div class="documento-texto">${dadosRetorno.documento}</div>
                    //         </div>`;
                    // código abaixo muda o fundo do navegador para destacar a folha branca       
                    document.body.style.display = "block";
                    document.body.style.backgroundColor = "#e5e7eb";
                    // document.body.style.padding = "20px";
                }
                else {
                    divResultado.innerHTML = `<strong>Erro no servidor:</strong>${dadosRetorno.mensagem}`;
                    divResultado.className = 'ERROR';
                }
            }
            catch (erro) {
                divResultado.innerHTML = `<strong>Erro no servidor:</strong>Não foi possível falar com o Back-end. O ser arquivo app.py está rodando no terminal?`;
                divResultado.className = 'ERROR';
                divResultado.classList.remove('hidden');

            }



        });




    </script>
